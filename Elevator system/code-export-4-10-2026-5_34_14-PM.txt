buildings [color:blue]{
  id serial pk
  name varchar(100)
  address varchar(255)
  city varchar(50)
  pincode varchar(10)
  total_floors int
}

floors [color:blue]{
  id serial pk
  floor_number int
  label varchar(20)

  building_id int fk
}

elevator_shafts [color:blue]{
  id serial pk
  shaft_label varchar(20)

  building_id int fk
}

elevators [color:blue]{
  id serial pk
  label varchar(20)
  capacity_kg int
  installed_date date

  shaft_id int fk
}

elevator_floor_service [color:blue]{
  id serial pk

  elevator_id int fk
  floor_id int fk
}

elevator_status_log [color:blue]{
  id serial pk
  status enum('idle','moving_up','moving_down','doors_open','maintenance','out_of_service')
  recorded_at timestamp

  elevator_id int fk
}

floor_requests [color:blue]{
  id serial pk
  direction enum('up','down')
  status enum('pending','assigned','completed','cancelled')
  requested_at timestamp

  floor_id int fk
}

ride_assignments [color:blue]{
  id serial pk
  assigned_at timestamp

  request_id int fk
  elevator_id int fk
}

ride_logs [color:blue]{
  id serial pk
  passenger_count int
  started_at timestamp
  ended_at timestamp

  assignment_id int fk
  to_floor_id int fk
}

maintenance_records [color:blue]{
  id serial pk
  type enum('scheduled','emergency','inspection')
  description text
  started_at timestamp
  ended_at timestamp
  technician_name varchar(100)
  status enum('scheduled','in_progress','completed')

  elevator_id int fk
}

buildings.id < floors.building_id
buildings.id < elevator_shafts.building_id

elevator_shafts.id < elevators.shaft_id

floors.id  < floor_requests.floor_id
floors.id  < elevator_floor_service.floor_id
floors.id  < ride_logs.to_floor_id

elevators.id < elevator_status_log.elevator_id
elevators.id < elevator_floor_service.elevator_id
elevators.id < ride_assignments.elevator_id
elevators.id < maintenance_records.elevator_id

floor_requests.id < ride_assignments.request_id

ride_assignments.id < ride_logs.assignment_id
